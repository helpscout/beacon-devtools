## Installation

```
npm install --save-dev @helpscout/beacon-devtools
```

## Usage

```jsx
import React from 'react'
import BeaconDevTools from '@helpscout/beacon-devtools'

class MyApp extends React.Component {
  render() {
    return (
      <div>
        ...
        <BeaconDevTools />
      </div>
    )
  }
}
```
